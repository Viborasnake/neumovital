document.addEventListener('DOMContentLoaded', function() {
    document.querySelectorAll('.sequence-container').forEach(container => {
        const img = container.querySelector('img');
        const totalFrames = parseInt(container.getAttribute('data-frames'));
        const prefix = container.getAttribute('data-prefix');
        const suffix = container.getAttribute('data-suffix');
        let currentFrame = 1;
        let interval;
        
        // Preload frames silently
        for(let i = 1; i <= totalFrames; i++) {
            let num = i.toString().padStart(3, '0');
            let preImg = new Image();
            preImg.src = prefix + num + suffix;
        }

        container.addEventListener('mouseenter', () => {
            if(interval) clearInterval(interval);
            interval = setInterval(() => {
                currentFrame++;
                if(currentFrame > totalFrames) currentFrame = 1;
                let num = currentFrame.toString().padStart(3, '0');
                img.src = prefix + num + suffix;
            }, 60); // approx 16 fps
        });

        container.addEventListener('mouseleave', () => {
            if(interval) clearInterval(interval);
            currentFrame = 1;
            img.src = prefix + '001' + suffix;
        });
    });
});
